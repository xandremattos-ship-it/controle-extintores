rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function userPath(userId) {
      return /databases/$(database)/documents/users/$(userId);
    }

    function profileExists() {
      return signedIn() && exists(userPath(request.auth.uid));
    }

    function profile() {
      return get(userPath(request.auth.uid)).data;
    }

    function activeProfile(data) {
      return data.get('active', false) == true
        && data.get('status', 'ATIVO') == 'ATIVO';
    }

    function activeUser() {
      return profileExists() && activeProfile(profile());
    }

    function ownershipPath() {
      return /databases/$(database)/documents/system/ownership;
    }

    function ownershipExists() {
      return exists(ownershipPath());
    }

    function ownership() {
      return get(ownershipPath()).data;
    }

    function isOwner() {
      return activeUser()
        && ownershipExists()
        && ownership().ownerUid == request.auth.uid
        && profile().role == 'PROPRIETARIO'
        && profile().get('isOwner', false) == true;
    }

    function isAdmin() {
      return activeUser()
        && profile().role == 'ADMINISTRADOR';
    }

    function hasAdministrativeAuthority() {
      return isOwner() || isAdmin();
    }

    function isInspector() {
      return activeUser() && profile().role == 'INSPETOR';
    }

    function isViewer() {
      return activeUser() && profile().role == 'VISUALIZADOR';
    }

    function hasAllUsinasAccess() {
      return isOwner() || isAdmin();
    }

    function sameUsina(usina) {
      return activeUser()
        && profile().usina is string
        && profile().usina == usina;
    }

    function validSelfRegistration(userId) {
      return signedIn()
        && request.auth.uid == userId
        && request.resource.data.keys().hasAll([
          'uid','name','email','role','usina','status','active',
          'requestedAt','createdAt','updatedAt'
        ])
        && request.resource.data.keys().hasOnly([
          'uid','name','email','role','usina','status','active',
          'requestedAt','createdAt','updatedAt'
        ])
        && request.resource.data.uid == request.auth.uid
        && request.resource.data.name is string
        && request.resource.data.name.size() >= 3
        && request.resource.data.name.size() <= 100
        && request.auth.token.email is string
        && request.resource.data.email == request.auth.token.email
        && request.resource.data.role == ''
        && request.resource.data.usina == ''
        && request.resource.data.status == 'PENDENTE'
        && request.resource.data.active == false
        && request.resource.data.requestedAt == request.time
        && request.resource.data.createdAt == request.time
        && request.resource.data.updatedAt == request.time;
    }

    function validOwnershipDocument() {
      return signedIn()
        && request.resource.data.keys().hasAll([
          'ownerUid','ownerEmail','hierarchyVersion','createdAt','updatedAt'
        ])
        && request.resource.data.keys().hasOnly([
          'ownerUid','ownerEmail','hierarchyVersion','createdAt','updatedAt'
        ])
        && request.resource.data.ownerUid == request.auth.uid
        && request.resource.data.ownerEmail is string
        && request.resource.data.hierarchyVersion == 1
        && request.resource.data.createdAt == request.time
        && request.resource.data.updatedAt == request.time;
    }

    function canClaimOwnership() {
      return signedIn()
        && !ownershipExists()
        && validOwnershipDocument()
        && (
          (
            profileExists()
            && activeUser()
            && profile().role in ['ADMINISTRADOR','PROPRIETARIO']
          )
          || (
            getAfter(userPath(request.auth.uid)).data.role == 'PROPRIETARIO'
            && getAfter(userPath(request.auth.uid)).data.get('isOwner', false) == true
          )
        );
    }

    function bootstrapOwnerProfile(userId) {
      return signedIn()
        && request.auth.uid == userId
        && !ownershipExists()
        && getAfter(ownershipPath()).data.ownerUid == request.auth.uid
        && request.resource.data.uid == request.auth.uid
        && request.resource.data.role == 'PROPRIETARIO'
        && request.resource.data.usina == 'TODAS'
        && request.resource.data.isOwner == true
        && request.resource.data.status == 'ATIVO'
        && request.resource.data.active == true;
    }

    function promoteLegacyOwner(userId) {
      return signedIn()
        && request.auth.uid == userId
        && resource.data.role in ['ADMINISTRADOR','PROPRIETARIO']
        && activeProfile(resource.data)
        && getAfter(ownershipPath()).data.ownerUid == request.auth.uid
        && request.resource.data.role == 'PROPRIETARIO'
        && request.resource.data.usina == 'TODAS'
        && request.resource.data.isOwner == true
        && request.resource.data.status == 'ATIVO'
        && request.resource.data.active == true
        && request.resource.data.diff(resource.data).affectedKeys().hasOnly([
          'role','usina','isOwner','hierarchyVersion','updatedAt'
        ]);
    }

    function validOwnerCreatedUser() {
      return request.resource.data.role in ['ADMINISTRADOR','INSPETOR','VISUALIZADOR']
        && request.resource.data.get('isOwner', false) == false
        && request.resource.data.status == 'ATIVO'
        && request.resource.data.active == true
        && (
          (
            request.resource.data.role == 'ADMINISTRADOR'
            && request.resource.data.usina == 'TODAS'
          )
          || (
            request.resource.data.role in ['INSPETOR','VISUALIZADOR']
            && request.resource.data.usina is string
            && request.resource.data.usina.size() > 0
          )
        );
    }

    function validAdminCreatedUser() {
      return request.resource.data.role in ['INSPETOR','VISUALIZADOR']
        && request.resource.data.get('isOwner', false) == false
        && request.resource.data.status == 'ATIVO'
        && request.resource.data.active == true
        && request.resource.data.usina is string
        && request.resource.data.usina.size() > 0;
    }

    function validOwnerUpdate(userId) {
      return isOwner()
        && (
          (
            userId == request.auth.uid
            && request.resource.data.role == 'PROPRIETARIO'
            && request.resource.data.usina == 'TODAS'
            && request.resource.data.isOwner == true
            && request.resource.data.status == 'ATIVO'
            && request.resource.data.active == true
          )
          || (
            userId != ownership().ownerUid
            && request.resource.data.role in ['ADMINISTRADOR','INSPETOR','VISUALIZADOR']
            && request.resource.data.get('isOwner', false) == false
          )
        );
    }

    function validAdminUpdate(userId) {
      return isAdmin()
        && userId != request.auth.uid
        && resource.data.role in ['INSPETOR','VISUALIZADOR']
        && request.resource.data.role in ['INSPETOR','VISUALIZADOR']
        && request.resource.data.get('isOwner', false) == false
        && request.resource.data.uid == resource.data.uid
        && request.resource.data.email == resource.data.email
        && request.resource.data.status in ['ATIVO','BLOQUEADO']
        && request.resource.data.active == (request.resource.data.status == 'ATIVO');
    }

    function validOwnNotificationState(userId) {
      return activeUser()
        && request.auth.uid == userId
        && request.resource.data.keys().hasAll(['uid','readIds','updatedAt'])
        && request.resource.data.keys().hasOnly(['uid','readIds','updatedAt','prunedAt'])
        && request.resource.data.uid == request.auth.uid
        && request.resource.data.readIds is list
        && request.resource.data.readIds.size() <= 500;
    }

    match /system/{documentId} {
      allow get: if signedIn();
      allow list: if false;
      allow create: if documentId == 'ownership' && canClaimOwnership();
      allow update, delete: if documentId == 'ownership' && isOwner();
    }

    match /users/{userId} {
      allow get: if signedIn()
        && (
          request.auth.uid == userId
          || isOwner()
          || (
            isAdmin()
            && resource.data.get('status', 'ATIVO') in ['ATIVO','BLOQUEADO']
          )
        );

      allow list: if isOwner()
        || (
          isAdmin()
          && resource.data.get('status', 'ATIVO') in ['ATIVO','BLOQUEADO']
        );

      allow create: if validSelfRegistration(userId)
        || bootstrapOwnerProfile(userId)
        || (isOwner() && validOwnerCreatedUser())
        || (isAdmin() && validAdminCreatedUser());

      allow update: if promoteLegacyOwner(userId)
        || validOwnerUpdate(userId)
        || validAdminUpdate(userId);

      allow delete: if (
          isOwner()
          && userId != ownership().ownerUid
        )
        || (
          isAdmin()
          && userId != request.auth.uid
          && resource.data.role in ['INSPETOR','VISUALIZADOR']
        );
    }

    match /adminAudit/{auditId} {
      allow read: if isOwner();
      allow create: if hasAdministrativeAuthority()
        && request.resource.data.keys().hasAll([
          'actorUid','actorName','actorRole','action','targetUid',
          'targetName','details','createdAt'
        ])
        && request.resource.data.keys().hasOnly([
          'actorUid','actorName','actorRole','action','targetUid',
          'targetName','details','createdAt'
        ])
        && request.resource.data.actorUid == request.auth.uid
        && request.resource.data.actorRole == profile().role
        && request.resource.data.createdAt == request.time;
      allow update, delete: if false;
    }

    match /notificationStates/{userId} {
      allow get: if activeUser() && request.auth.uid == userId;
      allow create, update: if validOwnNotificationState(userId);
      allow delete: if activeUser() && request.auth.uid == userId;
    }

    match /settings/{document=**} {
      allow read: if activeUser();
      allow write: if hasAdministrativeAuthority();
    }

    match /extinguishers/{extinguisherId} {
      allow read: if hasAllUsinasAccess() || sameUsina(resource.data.usina);

      allow create: if hasAllUsinasAccess()
        || (
          isInspector()
          && request.resource.data.usina == profile().usina
        );

      allow update: if hasAllUsinasAccess()
        || (
          isInspector()
          && resource.data.usina == profile().usina
          && request.resource.data.usina == profile().usina
        );

      allow delete: if hasAllUsinasAccess()
        || (
          isInspector()
          && resource.data.usina == profile().usina
        );
    }

    match /inspections/{inspectionId} {
      allow read: if hasAllUsinasAccess() || sameUsina(resource.data.usina);

      allow create: if hasAllUsinasAccess()
        || (
          isInspector()
          && request.resource.data.usina == profile().usina
        );

      allow update: if hasAllUsinasAccess()
        || (
          isInspector()
          && resource.data.usina == profile().usina
          && request.resource.data.usina == profile().usina
        );

      allow delete: if hasAllUsinasAccess();
    }

    match /appData/{document=**} {
      allow read: if hasAdministrativeAuthority();
      allow write: if false;
    }
  }
}
